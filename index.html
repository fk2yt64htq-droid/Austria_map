function submitFeedback() { const feedbackInput = document.getElementById("feedbackText"); if (!feedbackInput) { console.error("Елемент з id 'feedbackText' не знайдено на сторінці!"); return; } const text = feedbackInput.value.trim(); // Визначаємо поточну мову (захист від помилок, якщо змінна lang не ініціалізована) const currentLang = (typeof lang !== 'undefined' && lang) ? lang : 'uk'; if (!text) { alert(currentLang === "uk" ? "Будь ласка, введіть текст пропозиції." : "Please enter your suggestion."); return; } // Безпечно збираємо дані з Telegram WebApp let userId = 0; let username = ""; let firstName = "Водій"; if (typeof tg !== 'undefined' && tg && tg.initDataUnsafe && tg.initDataUnsafe.user) { userId = tg.initDataUnsafe.user.id || 0; username = tg.initDataUnsafe.user.username || ""; firstName = tg.initDataUnsafe.user.first_name || "Водій"; } // Надсилання запиту на сервер fetch(`${API_URL}/feedback`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ user_id: userId, username: username, first_name: firstName, text: text }) }) .then(res => { if (!res.ok) { throw new Error('Помилка сервера: ' + res.status); } return res.json(); }) .then(data => { // Успішно надіслано alert(currentLang === "uk" ? "Дякуємо за пропозицію! Вашу ідею збережено." : "Thank you for your suggestion! Your idea has been saved."); // Очищаємо поле тексту feedbackInput.value = ""; // Закриваємо модальне вікно (переконайся, що назва функції збігається з твоєю) if (typeof closeFeedbackModal === 'function') { closeFeedbackModal(); } }) .catch(err => { console.error("Критична помилка відправки:", err); alert(currentLang === "uk" ? "Не вдалося надіслати. Спробуйте пізніше." : "Failed to send. Please try again later."); }); }
